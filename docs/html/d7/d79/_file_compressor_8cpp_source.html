58void FileCompressor::do_Fano_Algorithm(std::optional<std::span<std::pair<uint8_t, size_t>>> vec, size_t vec_sum) {
59
60 if (!vec) {
61 if (occurrences.size() <= 1) {
62 supplement_codes(occurrences, 0);
63 return;
64 }
65 vec = std::span(occurrences);
66 vec_sum = occur_sum;
67 }
68 auto occur = *vec;
69
70 if (occur.size() == 1) return;
71
72 size_t sum_prev = 0, sum_cur = 0, i = 0;
73
74 while (i < occur.size() && sum_cur + occur[i].second <= vec_sum / 2) {
75 sum_cur += occur[i++].second;
76 }
77 size_t left_sum = sum_cur;

238void FileCompressor::load_archived(std::fstream& file) {
239 uint8_t tableSize;
240 file.read(reinterpret_cast<char*>(&tableSize), sizeof(tableSize));
241 codes.clear();
242
243 for (int i = 0; i < tableSize; ++i) {
244 uint8_t symbol, bitCount;
245 file.read(reinterpret_cast<char*>(&symbol), 1);
246 file.read(reinterpret_cast<char*>(&bitCount), 1);
247
248 std::vector<bool> bits;
249 bits.reserve(bitCount);
250 int bitsRead = 0;
251 while (bitsRead < bitCount) {
252 uint8_t byte;
253 file.read(reinterpret_cast<char*>(&byte), 1);
254 for (int b = 0; b < 8 && bitsRead < bitCount; ++b) {
255 bool bit = (byte >> (7 - b)) & 1;
256 bits.push_back(bit);
257 bitsRead++;
258 }
259 }
260 codes[symbol] = bits;
261 }
